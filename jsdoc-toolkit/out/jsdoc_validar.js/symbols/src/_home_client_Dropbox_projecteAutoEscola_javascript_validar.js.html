 10 var contaErrores = 0;

 12 /**
 13  * @param valor Es el caracter que coge del 'input' telefono
 14  * @desc Funcion creada para validar que el caracter introducido es un entero, realiza el parseo a integer.
 15  * @returns Devuelve el valor (el numero entero) en el caso que se pueda parsear, en caso contrario devuelve el string vacio
 16  */
 17 function validarEntero(valor) {
 18 	//intento convertir a entero.
 19 	//si era un entero no le afecta, si no lo era lo intenta convertir
 20 	valor = parseInt(valor);
 21 
 22 	//Compruebo si es un valor numérico
 23 	if (isNaN(valor)) {
 24 		//entonces (no es numero) devuelvo el valor cadena vacia
 25 		return ""
 26 	} else {
 27 		//En caso contrario (Si era un número) devuelvo el valor
 28 		return valor
 29 	}
 30 }

 32 /**
 33  * @param campo Es el nombre del elemento 'section' correspondiente al 'input' que esta validando en ese momento donde se mostrará el error
 34  * @param texto Es el texto que recoge para mostrarlo en el navegador como error
 35  * @desc Funcion para crear un mensaje de error en el navegador cuando el 'input' no supere la validacion,
 36  * primero comprueba que no haya ya algun 'hijo' en el campo (ya haya mostrado el error anteriormente), si no hay error lo crea por DOM.
 37  */
 38 function mostraError(campo, texto) {
 39 	var element = document.getElementById(campo);
 40 
 41 	var conta = element.childNodes.length;
 42 
 43 	if (conta == 0) {
 44 		var etiqueta = document.createElement("h6");
 45 		var node = document.createTextNode(texto);
 46 		etiqueta.appendChild(node);
 47 
 48 		element.appendChild(etiqueta);
 49 	}
 50 }

 52 /**
 53  * @param campo Es el nombre del elemento 'section' correspondiente al 'input' que esta validando en ese momento donde eliminará el error
 54  * @desc Funcion para eliminar el mensaje de error en el navegador cuando el 'input' YA supere la validacion,
 55  */
 56 function suprimixError(campo) {
 57 
 58 	var element = document.getElementById(campo);
 59 
 60 	element.removeChild(element.firstChild);
 61 
 62 }
 63 
 64 /**
 65  *
 66  * @param {Object} obj Recibe como parametro el objeto, en este caso el 'input' correspondiente al campo que se esta validando.
 67  * @desc Dependiendo de que 'input' estemos validando entrará en una condicion según el nombre del objeto,
 68  * cada campo a validar tiene su propia expresión regular que valida que esté correctamente escrito,
 69  * si no pasa la validación llama a la funcion 'mostraError' con los parametros correspondientes para que muestre el error en el navegador,
 70  * (también desactiva el boton 'Modificar' en el caso que estemos validando los campos en la pagina de modificación, ya que este validador
 71  * es usado tanto para dar de alta como para modificar alumnos y profesores)
 72  * en el caso que supere la validación, elimina el error que anteriormente habia mostrado en el navegador, y aumenta la variable
 73  * 'contaErrores' en uno.
 74  * @returns Devuelve true si no hay ningun error, en el momento que haya un error devuelve false.
 75  */
 76 function valida_envia(obj) {
 77 	erDNI = /^[0-9]{8}[A-Z]$/;
 78 	erNombre = /^[a-zA-ZñÑáÁéÉíÍóÓúÚüÜ ]{3,20}$/;
 79 
 80 	//alert(obj.name);
 81 	if (obj.name == "login") {
 82 		//valido el login
 83 		if (erNombre.test(document.formulario.login.value) == false) {
 84 			mostraError("divLogin", "Login: Min. 3 caracteres, Max. 20. Sin digitos");
 85 			return false;
 86 		} else {
 87 			suprimixError("divLogin");
 88 			contaErrores++;
 89 			//alert(contaErrores);
 90 		}
 91 	}
 92 	if (obj.name == "password") {
 93 		//valido el password
 94 		if (erNombre.test(document.formulario.password.value) == false) {
 95 			mostraError("divPassword", "Password: Min. 3 caracteres, Max. 20. Sin digitos");
 96 			document.getElementById("botoModifica").disabled = true;
 97 			return false;
 98 		} else {
 99 			suprimixError("divPassword");
100 			contaErrores++;
101 
102 		}
103 	}
104 	if (obj.name == "nombre") {
105 		//valido el nombre
106 		if (erNombre.test(document.formulario.nombre.value) == false) {
107 			mostraError("divNombre", "Nombre: Min. 3 caracteres, Max. 20. Sin digitos");
108 			document.getElementById("botoModifica").disabled = true;
109 			return false;
110 		} else {
111 			suprimixError("divNombre");
112 			contaErrores++;
113 
114 		}
115 	}
116 	if (obj.name == "apellido") {
117 		//valido el apellido
118 		if (erNombre.test(document.formulario.apellido.value) == false) {
119 			mostraError("divApellido", "Apellido: Min. 3 caracteres, Max. 20. Sin digitos");
120 			document.getElementById("botoModifica").disabled = true;
121 			errorForm = true;
122 			return false;
123 		} else {
124 			suprimixError("divApellido");
125 			contaErrores++;
126 
127 		}
128 	}
129 	if (obj.name == "dni") {
130 		//valido el dni
131 		if (erDNI.test(document.formulario.dni.value) == false) {
132 			mostraError("divDNI", "DNI: 8 digitos y 1 letra mayuscula");
133 			document.getElementById("botoModifica").disabled = true;
134 			errorForm = true;
135 			return false;
136 		} else {
137 			//valido que sea un DNI valido
138 			dni = document.formulario.dni.value;
139 			numero = dni.substr(0, dni.length - 1);
140 			let = dni.substr(dni.length - 1, 1);
141 			numero = numero % 23;
142 			letra = 'TRWAGMYFPDXBNJZSQVHLCKET';
143 			letra = letra.substring(numero, numero + 1);
144 			if (letra != let) {
145 				alert('Dni erroneo');
146 				return false;
147 			} else {
148 				suprimixError("divDNI");
149 				contaErrores++;
150 
151 			}
152 
153 		}
154 	}
155 	if (obj.name == "email") {
156 		if (document.formulario.email.value.length == 0) {
157 			mostraError("divEmail", "eMail: Tiene que escribir su email");
158 			document.formulario.email.focus();
159 			return false;
160 		} else {
161 			correo = document.formulario.email.value;
162 			if (!/^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,4})+$/.test(correo)) {
163 				mostraError("divEmail", "La direccion de email es incorrecta.");
164 				document.getElementById("botoModifica").disabled = true;
165 				errorForm = true;
166 				document.formulario.email.focus();
167 				return false;
168 			} else {
169 				suprimixError("divEmail");
170 				contaErrores++;
171 
172 			}
173 		}
174 	}
175 	if (obj.name == "telefono") {
176 		//valido el telefono. tiene que ser entero y con 9 digitos
177 		telefono = document.formulario.telefono.value;
178 		telefono = validarEntero(telefono);
179 		document.formulario.telefono.value = telefono;
180 		if (telefono == "") {
181 			mostraError("divTelefono", "Tiene que introducir digitos en el telefono");
182 			document.formulario.telefono.focus();
183 			return false;
184 		} else {
185 			if (document.formulario.telefono.value.length != 9) {
186 				mostraError("divTelefono", "El numero tiene que tener 9 digitos");
187 				document.getElementById("botoModifica").disabled = true;
188 				errorForm = true;
189 				document.formulario.telefono.focus();
190 				return false;
191 			} else {
192 				suprimixError("divTelefono");
193 				contaErrores++;
194 
195 			}
196 		}
197 	}
198 	/*
199 	 * @global errores Variable global utilizada para comprobar si tiene hijos por DOM
200 	 * @global erroress Variable global utilizada como contador para contar cuantos hijos tiene la variable 'errores'	 *
201 	 */
202 	var errores = document.getElementById("divErrores");
203 	var erroress = 0;
204 	/**
205 	 * @desc Recorro el elemento 'errores' para comprobar si tiene hijos, si tiene, aumento en 1 el contador 'erroress'
206 	 */
207 	for (var i = 0; i < errores.childNodes.length; i++) {
208 		if (errores.childNodes[i].childNodes.length > 0) {
209 			erroress++;
210 		}
211 	}
212 	/**
213 	 * @desc Si 'erroress' es mayor que 0, (hay algun error), desactiva el 'botonModifica', en caso que no haya errores, activa el boton.
214 	 */
215 	if (erroress > 0) {
216 		document.getElementById("botoModifica").disabled = true;
217 	} else {
218 		document.getElementById("botoModifica").disabled = false;
219 
220 	}
221 
222 	return true;
223 }

225 /**
226  * @desc Funcion creada para enviar el formulario, cuando todos los campos esten validados correctamente,
227  * la variable 'tablaModifica' esta creada para comprobar si existe, si existe significa que estamos validando el formulario de modificacion de usuarios
228  * si no existe, estamos validando el formulario de alta de usuarios.
229  * Para enviar los datos al formualario de alta, tiene que ser el valor de 'contaErrores' igual a 7, ya que hay 7 campos a validar
230  * y cada uno de ellos ha superado la validacion, en caso que no sea igual a 7, muestra un error para que se verifiquen los datos introducidos.
231  * Si existe la 'tablaModifica' entonces el envio del formulario dependerá de si el botón de 'Modificar' esta activado o no.
232  */
233 function enviar() {
234 	var tablaModifica = document.getElementById("tablaModifica");
235 
236 	if (tablaModifica == null) {
237 		if (contaErrores == 7) {
238 			document.formulario.submit();
239 		} else {
240 			alert("Revise los campos!");
241 		}
242 	} else {
243 		document.formulario.submit();
244 	}
245 
246 }
