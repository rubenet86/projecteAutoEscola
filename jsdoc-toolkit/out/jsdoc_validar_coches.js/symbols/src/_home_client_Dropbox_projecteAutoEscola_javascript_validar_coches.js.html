 11 
 12 var contaErrores = 0;
 13 
 14 /**
 15  * @param campo Es el nombre del elemento 'section' correspondiente al 'input' que esta validando en ese momento donde se mostrará el error
 16  * @param texto Es el texto que recoge para mostrarlo en el navegador como error
 17  * @desc Funcion para crear un mensaje de error en el navegador cuando el 'input' no supere la validacion,
 18  * primero comprueba que no haya ya algun 'hijo' en el campo (ya haya mostrado el error anteriormente), si no hay error lo crea por DOM.
 19  */
 20 function mostraError(campo, texto) {
 21 	var element = document.getElementById(campo);
 22 	var conta = element.childNodes.length;
 23 
 24 	if (conta == 0) {
 25 		var etiqueta = document.createElement("h6");
 26 		var node = document.createTextNode(texto);
 27 		etiqueta.appendChild(node);
 28 
 29 		element.appendChild(etiqueta);
 30 	}
 31 }
 32 
 33 /**
 34  * @param campo Es el nombre del elemento 'section' correspondiente al 'input' que esta validando en ese momento donde eliminará el error
 35  * @desc Funcion para eliminar el mensaje de error en el navegador cuando el 'input' YA supere la validacion,
 36  */
 37 function suprimixError(campo) {
 38 
 39 	var element = document.getElementById(campo);
 40 	element.removeChild(element.firstChild);
 41 	//alert("nia");
 42 
 43 }
 44 
 45 /**
 46  *
 47  * @param {Object} obj Recibe como parametro el objeto, en este caso el 'input' correspondiente al campo que se esta validando.
 48  * @desc Dependiendo de que 'input' estemos validando entrará en una condicion según el nombre del objeto,
 49  * cada campo a validar tiene su propia expresión regular que valida que esté correctamente escrito,
 50  * si no pasa la validación llama a la funcion 'mostraError' con los parametros correspondientes para que muestre el error en el navegador,
 51  * (también desactiva el boton 'Modificar' en el caso que estemos validando los campos en la pagina de modificación, ya que este validador
 52  * es usado tanto para dar de alta como para modificar coches)
 53  * en el caso que supere la validación, elimina el error que anteriormente habia mostrado en el navegador, y aumenta la variable
 54  * 'contaErrores' en uno.
 55  * @returns Devuelve true si no hay ningun error, en el momento que haya un error devuelve false.
 56  */
 57 function valida_coche(obj) {
 58 	erMatricula = /^[0-9]{4}[A-Z]{3}$/;
 59 	erNombre = /^[a-zA-ZñÑáÁéÉíÍóÓúÚüÜ ]{3,20}$/;
 60 
 61 	//valido la matricula
 62 	if (obj.name == "matricula") {
 63 		if (erMatricula.test(document.formulario.matricula.value) == false) {
 64 			mostraError("divMatricula", "4 digitos y 3 letras MAYUSCULAS");
 65 			document.getElementById("botoModifica").disabled = true;
 66 			return false;
 67 		} else {
 68 			suprimixError("divMatricula");
 69 			contaErrores++;
 70 		}
 71 	}
 72 
 73 	//valido la marca
 74 	if (obj.name == "marca") {
 75 		if (erNombre.test(document.formulario.marca.value) == false) {
 76 			mostraError("divMarca", "Min. 3 caracteres, Max. 20. Sin digitos");
 77 			document.getElementById("botoModifica").disabled = true;
 78 
 79 			return false;
 80 		} else {
 81 			suprimixError("divMarca");
 82 			contaErrores++;
 83 
 84 		}
 85 	}
 86 
 87 	//valido la marca
 88 	if (obj.name == "modelo") {
 89 		if (erNombre.test(document.formulario.modelo.value) == false) {
 90 			mostraError("divModelo", "Min. 3 caracteres, Max. 20. Sin digitos");
 91 			document.getElementById("botoModifica").disabled = true;
 92 
 93 			return false;
 94 		} else {
 95 			suprimixError("divModelo");
 96 			contaErrores++;
 97 
 98 		}
 99 	}
100 	//valido el color
101 	if (obj.name == "color") {
102 		if (erNombre.test(document.formulario.color.value) == false) {
103 			mostraError("divColor", "Min. 3 caracteres, Max. 20. Sin digitos");
104 			document.getElementById("botoModifica").disabled = true;
105 
106 			return false;
107 		} else {
108 			suprimixError("divColor");
109 			contaErrores++;
110 
111 		}
112 	}
113 	/*
114 	 * @global errores Variable global utilizada para comprobar si tiene hijos por DOM
115 	 * @global erroress Variable global utilizada como contador para contar cuantos hijos tiene la variable 'errores'	 *
116 	 */
117 	var errores = document.getElementById("divErrores");
118 	var erroress = 0;
119 
120 	/**
121 	 * @desc Recorro el elemento 'errores' para comprobar si tiene hijos, si tiene, aumento en 1 el contador 'erroress'
122 	 */
123 	for (var i = 0; i < errores.childNodes.length; i++) {
124 		if (errores.childNodes[i].childNodes.length > 0) {
125 			erroress++;
126 		}
127 	}
128 
129 	/**
130 	 * @desc Si 'erroress' es mayor que 0, (hay algun error), desactiva el 'botonModifica', en caso que no haya errores, activa el boton.
131 	 */
132 	if (erroress > 0) {
133 		document.getElementById("botoModifica").disabled = true;
134 	} else {
135 		document.getElementById("botoModifica").disabled = false;
136 	}
137 
138 	return true;
139 }

141 /**
142  * @desc Funcion creada para enviar el formulario, cuando todos los campos esten validados correctamente,
143  * la variable 'tablaModifica' esta creada para comprobar si existe, si existe significa que estamos validando el formulario de modificacion de coches
144  * si no existe, estamos validando el formulario de alta de coches.
145  * Para enviar los datos al formualario de alta, tiene que ser el valor de 'contaErrores' igual a 4, ya que hay 4 campos a validar
146  * y cada uno de ellos ha superado la validacion, en caso que no sea igual a 4, muestra un error para que se verifiquen los datos introducidos.
147  * Si existe la 'tablaModifica' entonces el envio del formulario dependerá de si el botón de 'Modificar' esta activado o no.
148  */
149 function enviar_coche() {
150 	var tablaAlta = document.getElementById("tablaModifica");
151 	if (tablaAlta == null) {
152 		if (contaErrores == 4) {
153 			document.formulario.submit();
154 		} else {
155 			alert("Revise los campos!");
156 		}
157 	} else {
158 		document.formulario.submit();
159 	}

161 }
